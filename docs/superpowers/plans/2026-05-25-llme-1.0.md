# LLMe 1.0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship LLMe 1.0 — a Rust CLI + shell-hook coordinator that materializes per-scope AI-agent config and injects an ICM MCP across a multi-host setup.

**Architecture:** Single Rust binary `llme`. Declarative TOML config in a user-owned git repo. Scope matchers (network/host/user/project) → tag set → bundle layering → materialized cache dir → agent env vars via shell hook. Pluggable per-agent adapters; Claude Code is the 1.0 reference. ICM MCP topology built in.

**Tech Stack:** Rust 1.95+ stable, edition 2024, `tokio` 1.52, `clap` 4.6, `serde` 1, `toml` 1.1, `tracing` 0.1 / `tracing-subscriber` 0.3, `assert_cmd` 2 + `insta` 1.47 for tests, shell-out to `git`.

```rust
use super::Config;
use thiserror::Error;

#[derive(Debug, Error)]
pub enum ValidateError {
    #[error("duplicate scope id: {0}")]
    DuplicateScopeId(String),
    #[error("bundle {0} has no tags")]
    BundleNoTags(String),
}

impl Config {
    pub fn validate(&self) -> Result<(), ValidateError> {
        let mut seen = std::collections::HashSet::new();
        let ids = self.scope.network.iter().map(|s| &s.id)
            .chain(self.scope.host.iter().map(|s| &s.id))
            .chain(self.scope.user.iter().map(|s| &s.id))
            .chain(self.scope.project.iter().map(|s| &s.id));
        for id in ids {
            if !seen.insert(id.clone()) {
                return Err(ValidateError::DuplicateScopeId(id.clone()));
            }
        }
        for b in &self.bundle {
            if b.tags.is_empty() {
                return Err(ValidateError::BundleNoTags(b.name.clone()));
            }
        }
        Ok(())
    }
}
```

```rust
use crate::config::*;

pub struct Env {
    pub hostname: String,
    pub user: String,
    pub cwd: String,
    pub gateway_mac: Option<String>,
}
impl Env {
    pub fn empty() -> Self { Self { hostname: String::new(), user: String::new(), cwd: String::new(), gateway_mac: None } }
    pub fn detect() -> Self {
        Self {
            hostname: hostname().unwrap_or_default(),
            user: std::env::var("USER").unwrap_or_default(),
            cwd: std::env::current_dir().ok().and_then(|p| p.to_str().map(String::from)).unwrap_or_default(),
            gateway_mac: None, // filled in by network module
        }
    }
}

fn hostname() -> Option<String> {
    std::process::Command::new("hostname").output().ok()
        .and_then(|o| String::from_utf8(o.stdout).ok())
        .map(|s| s.trim().to_string())
}

pub fn matches_network(s: &NetworkScope, env: &Env) -> bool {
    if let Some(m) = &s.r#match.gateway_mac {
        if env.gateway_mac.as_deref() != Some(m.as_str()) { return false; }
    }
    // ssid/cidr left for later tasks
    s.r#match.gateway_mac.is_some()
}
pub fn matches_host(s: &HostScope, env: &Env) -> bool {
    s.r#match.hostname.as_deref().map(|h| h == env.hostname).unwrap_or(false)
}
pub fn matches_user(s: &UserScope, env: &Env) -> bool {
    s.r#match.user.as_deref().map(|u| u == env.user).unwrap_or(false)
}
pub fn matches_project(s: &ProjectScope, env: &Env) -> bool {
    if let Some(p) = &s.r#match.path_prefix {
        let p = shellexpand::tilde(p).into_owned();
        if env.cwd.starts_with(&p) { return true; }
    }
    if let Some(m) = &s.r#match.marker_file {
        let mut cur = std::path::PathBuf::from(&env.cwd);
        loop {
            if cur.join(m).exists() { return true; }
            if !cur.pop() { break; }
        }
    }
    false
}
```

```rust
pub mod matcher;
pub use matcher::Env;

use crate::config::Config;
use std::collections::BTreeSet;

#[derive(Debug)]
pub struct ActiveScope {
    pub id: String,
    pub kind: &'static str,
    pub tags: Vec<String>,
}

#[derive(Debug)]
pub struct ActiveScopes {
    pub scopes: Vec<ActiveScope>,
    pub tags: BTreeSet<String>,
}

pub fn evaluate(cfg: &Config, env: &Env) -> ActiveScopes {
    let mut scopes = Vec::new();
    for s in &cfg.scope.network {
        if matcher::matches_network(s, env) { scopes.push(ActiveScope { id: s.id.clone(), kind: "network", tags: s.tags.clone() }); }
    }
    for s in &cfg.scope.host {
        if matcher::matches_host(s, env) { scopes.push(ActiveScope { id: s.id.clone(), kind: "host", tags: s.tags.clone() }); }
    }
    for s in &cfg.scope.user {
        if matcher::matches_user(s, env) { scopes.push(ActiveScope { id: s.id.clone(), kind: "user", tags: s.tags.clone() }); }
    }
    for s in &cfg.scope.project {
        if matcher::matches_project(s, env) { scopes.push(ActiveScope { id: s.id.clone(), kind: "project", tags: s.tags.clone() }); }
    }
    let tags: BTreeSet<String> = scopes.iter().flat_map(|s| s.tags.iter().cloned()).collect();
    ActiveScopes { scopes, tags }
}
```

```rust
pub fn detect_gateway_mac() -> Option<String> {
    // macOS: `route -n get default` + `arp -n <gw>`
    // Linux: `ip route show default` + `ip neigh show <gw>`
    #[cfg(target_os = "macos")] {
        let gw = String::from_utf8(std::process::Command::new("route").args(["-n","get","default"]).output().ok()?.stdout).ok()?;
        let gw_ip = gw.lines().find_map(|l| l.trim().strip_prefix("gateway: ").map(str::trim).map(String::from))?;
        let arp = String::from_utf8(std::process::Command::new("arp").args(["-n", &gw_ip]).output().ok()?.stdout).ok()?;
        arp.split_whitespace().find(|t| t.contains(':') && t.len() == 17).map(String::from)
    }
    #[cfg(target_os = "linux")] {
        let r = String::from_utf8(std::process::Command::new("ip").args(["route","show","default"]).output().ok()?.stdout).ok()?;
        let gw_ip = r.split_whitespace().nth(2).map(String::from)?;
        let n = String::from_utf8(std::process::Command::new("ip").args(["neigh","show",&gw_ip]).output().ok()?.stdout).ok()?;
        n.split_whitespace().find(|t| t.contains(':') && t.len() == 17).map(String::from)
    }
    #[cfg(not(any(target_os = "macos", target_os = "linux")))] { None }
}
```

```rust
pub mod agents_md;
use std::collections::BTreeMap;
use std::path::{Path, PathBuf};

pub struct BundleRef { pub name: String, pub path: PathBuf }

pub struct MergedManifest {
    pub agents_md: String,
    /// relative path → absolute source path (later bundles overwrite earlier)
    pub files: BTreeMap<PathBuf, PathBuf>,
}

pub fn merge(bundles: &[BundleRef]) -> anyhow::Result<MergedManifest> {
    let mut agents_parts = Vec::new();
    let mut files = BTreeMap::new();
    for b in bundles {
        let am = b.path.join("AGENTS.md");
        if am.exists() {
            agents_parts.push((b.name.clone(), std::fs::read_to_string(&am)?));
        }
        for sub in ["skills", "plugins", "hooks"] {
            let dir = b.path.join(sub);
            if !dir.exists() { continue; }
            walk(&dir, &dir, sub, &mut files)?;
        }
    }
    Ok(MergedManifest { agents_md: agents_md::concat(&agents_parts), files })
}

fn walk(root: &Path, dir: &Path, prefix: &str, out: &mut BTreeMap<PathBuf, PathBuf>) -> anyhow::Result<()> {
    for entry in std::fs::read_dir(dir)? {
        let entry = entry?;
        let p = entry.path();
        if p.is_dir() { walk(root, &p, prefix, out)?; }
        else {
            let rel = p.strip_prefix(root.parent().unwrap()).unwrap().to_path_buf();
            out.insert(rel, p);
        }
    }
    Ok(())
}
```

```rust
pub fn gc(cache_root: &std::path::Path, older_than: std::time::Duration) -> anyhow::Result<()> {
    let now = std::time::SystemTime::now();
    for entry in std::fs::read_dir(cache_root)? {
        let e = entry?;
        let p = e.path();
        if !p.is_dir() { continue; }
        let m = newest_mtime(&p)?;
        if now.duration_since(m).unwrap_or_default() > older_than {
            std::fs::remove_dir_all(&p)?;
        }
    }
    Ok(())
}
fn newest_mtime(dir: &std::path::Path) -> anyhow::Result<std::time::SystemTime> {
    let mut max = std::time::UNIX_EPOCH;
    for entry in walkdir::WalkDir::new(dir) {
        let e = entry?;
        let m = e.metadata()?.modified()?;
        if m > max { max = m; }
    }
    Ok(max)
}
```

```rust
use std::path::PathBuf;

pub fn pidfile() -> PathBuf {
    PathBuf::from(std::env::var("HOME").unwrap()).join(".local/state/llmenv/mcp-proxy.pid")
}
pub fn ensure_running(bind: &str) -> anyhow::Result<()> {
    let pf = pidfile();
    if pf.exists() {
        if let Ok(pid) = std::fs::read_to_string(&pf).map(|s| s.trim().to_string()) {
            if proc_alive(&pid) { return Ok(()); }
        }
    }
    std::fs::create_dir_all(pf.parent().unwrap())?;
    let child = std::process::Command::new("mcp-proxy")
        .args(["--port", bind.split(':').last().unwrap_or("7878"), "--", "icm", "mcp-server"])
        .stdin(std::process::Stdio::null())
        .stdout(std::process::Stdio::null())
        .stderr(std::process::Stdio::null())
        .spawn()?;
    std::fs::write(&pf, child.id().to_string())?;
    Ok(())
}
fn proc_alive(pid: &str) -> bool {
    std::process::Command::new("kill").args(["-0", pid]).status().map(|s| s.success()).unwrap_or(false)
}
```

```rust
use std::path::Path;
use std::time::{Duration, SystemTime};

pub fn state_path(state_dir: &Path) -> std::path::PathBuf { state_dir.join("sync.json") }

pub fn read_state(state_dir: &Path) -> anyhow::Result<Option<SystemTime>> {
    let p = state_path(state_dir);
    if !p.exists() { return Ok(None); }
    let s = std::fs::read_to_string(&p)?;
    let secs: u64 = s.trim().parse()?;
    Ok(Some(std::time::UNIX_EPOCH + Duration::from_secs(secs)))
}
pub fn write_state(state_dir: &Path, t: SystemTime) -> anyhow::Result<()> {
    std::fs::create_dir_all(state_dir)?;
    std::fs::write(state_path(state_dir), t.duration_since(std::time::UNIX_EPOCH)?.as_secs().to_string())?;
    Ok(())
}

pub fn maybe_pull(repo: &Path, state_dir: &Path, interval: Duration) -> anyhow::Result<()> {
    let now = SystemTime::now();
    if let Some(last) = read_state(state_dir)? {
        if now.duration_since(last).unwrap_or_default() < interval { return Ok(()); }
    }
    let _ = std::process::Command::new("git").args(["fetch"]).current_dir(repo).status();
    let _ = std::process::Command::new("git").args(["pull","--ff-only"]).current_dir(repo).status();
    write_state(state_dir, now)?;
    Ok(())
}

pub fn run() -> anyhow::Result<()> {
    let dir = crate::paths::config_dir()?;
    std::process::Command::new("git").arg("add").arg("-A").current_dir(&dir).status()?;
    std::process::Command::new("git").args(["commit","-m","llme sync"]).current_dir(&dir).status().ok();
    std::process::Command::new("git").arg("push").current_dir(&dir).status()?;
    Ok(())
}
```
